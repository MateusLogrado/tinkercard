                int pot1 = 0;
                int pot2 = 0;
                int res = 0;
                
                void setup()
                {
                  pinMode(A0, INPUT);
                  Serial.begin(9600);
                  pinMode(A1, INPUT);
                  pinMode(12, OUTPUT);
                
                  pot1 = 0;
                  pot2 = 0;
                  res = 0;
                }
                
                void loop()
                {
                  pot1 += analogRead(A0);
                
                  Serial.println("port1 ------------");
                  Serial.println(pot1);
                
                  pot2 = analogRead(A1);
                  Serial.println("port2 ------------");
                  Serial.println(pot2);
                  
                  if (pot1 > pot2) {
                    res += (pot1 + pot2);
                    Serial.println("res--------------------");
                    Serial.println(res);
                    digitalWrite(12, HIGH);
                  } else {
                    digitalWrite(12, LOW);
                  }
                  delay(1000); // Wait for 1000 millisecond(s)
                }
